## Now we can create some sample data to test networks
# We'll create some different line shapes then add noise to them

# Define number of samples
num_samples = 100000
x = torch.arange(0,10,0.1).unsqueeze(1)

# Initialize empty lists to store data
x_data = []
y_data_noisy = []
y_data_true = []

# Loop num_samples times
for _ in range(num_samples):
    # Generate random m and c values
    a = np.random.uniform(0,10)
    b = np.random.uniform(0,10)
    c = np.random.uniform(0,10)
    d = np.random.uniform(0,3)
    

    # Generate lines
    x = torch.arange(0, 10, 0.1).unsqueeze(1)
    i = torch.randint(0,6,(1,1))
    if i == 0:
        y_true = a * x + c  # straight line
    if i == 1:
        y_true = a*np.sin(d*x + c) + b  # sin wave
    if i == 2:
        y_true = torch.ones(100,1)*5*c  # flat line
    if i == 3:
        y_true = -a * x + c  # -ve straight line
    if i == 4:
        y_true = torch.ones(100,1)   # 'box' line
        y1 = torch.ones(30,1)*5*a
        y2 = torch.ones(40,1)*5*(a+5)
        y_true[:30] = y1[:30]
        y_true[30:70] = y2[:40]
        y_true[70:100] = y1[:30]
    elif i == 5:
        y_true = torch.ones(100,1)   # double 'box' line
        y1 = torch.ones(20,1)*5*a
        y2 = torch.ones(20,1)*5*b
        y3 = torch.ones(20,1)*5*c
        y_true[:20] = y1[:]
        y_true[20:40] = y2[:]
        y_true[40:60] = y1[:]
        y_true[60:80] = y3[:]
        y_true[80:100] = y1[:]

    # Add random noise
    noise = torch.randn(100,1)
    y_noisy = y_true + noise

    # Append to the lists
    x_data.append(x)
    y_data_noisy.append(y_noisy)
    y_data_true.append(y_true)

# Convert the lists to tensors
x_data = torch.stack(x_data, dim=0)
y_data_noisy = torch.stack(y_data_noisy, dim=0)
y_data_true = torch.stack(y_data_true, dim=0)

# Define a training function
def train_model(model, epochs, optimizer, loss_fn, train_in=train_in, train_out=train_out, test_in=test_in, test_out=test_out):
    model_train_start = timer()
    for epoch in tqdm(range(epochs)):

        model.train()  # train model

        y_pred = model(train_in)  # forward pass
        
        loss = loss_fn(y_pred, train_out)  # calculate loss

        optimizer.zero_grad()

        loss.backward()  # backpropagation

        optimizer.step()  # gradient descent

    # --------- Testing the model -------------

        model.eval()  # switch to testing mode

        with torch.inference_mode():  # inference mode

            test_pred = model(test_in)  # forward pass

            test_loss = loss_fn(test_pred, test_out)  # calculate loss

        # print out what's happening
        if epoch % int(epochs/10) == 0:
            print(f'Epoch: {epoch} | Loss: {loss:.4f} | Test loss: {test_loss:.4f}')
    model_train_end = timer()
    model_total_train = print_train_time(start=model_train_start,
                         end=model_train_end,
                         device=str(next(model.parameters()).device))  # returns current model_0 device
    return test_pred

# create a new model

class DeNoiserV1(nn.Module):
    def __init__(self, input_no, output_no, hidden_no):
        super().__init__()

        self.conv1 = nn.Conv1d(in_channels=input_no, out_channels=hidden_no, kernel_size=1)
        self.pool = nn.MaxPool1d(kernel_size=1)
        self.conv2 = nn.Conv1d(in_channels=hidden_no, out_channels=hidden_no, kernel_size=1)
        self.flat = nn.Flatten()
        self.fc1 = nn.Linear(in_features=hidden_no , out_features=output_no)
    
    def forward(self, x):
        x = self.conv1(x)
        x = self.conv2(x)
        x = self.pool(x)
        x = self.flat(x)
        x = self.fc1(x)
        x = x.view(-1, 100, 1)
        return x

model_1 = DeNoiserV1(100, 100, 8)

optimizer_1 = torch.optim.Adam(model_1.parameters(), lr=0.001)
loss_fn_1 = nn.L1Loss()

model_1.to(device)

# Save the model
from pathlib import Path

def Save_Model(model, name):

    MODEL_PATH = Path("models")  # define path
    MODEL_PATH.mkdir(parents=True,  # create if they don't exist
                     exist_ok=True)  # save in folder if it does exist
    MODEL_NAME = name + ".pth"
    MODEL_SAVE_PATH = MODEL_PATH / MODEL_NAME

    print(f"Saving model to {MODEL_PATH} as {MODEL_NAME}")
    torch.save(obj=model.state_dict(), f=MODEL_SAVE_PATH)
    
    return
